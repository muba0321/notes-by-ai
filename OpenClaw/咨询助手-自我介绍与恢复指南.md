# 清理
rm -rf "$BACKUP_DIR" "/tmp/openclaw-backup-${DATE}.tar.gz"
EOF

chmod +x /root/.openclaw/backup.sh

# 添加Cron：每天凌晨3点备份
(crontab -l 2>/dev/null; echo "0 3 * * * /root/.openclaw/backup.sh") | crontab -
